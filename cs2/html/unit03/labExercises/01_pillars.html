WebSite Code Files

A website consists of several different types of coding languages. The three most important are HTML, CSS, and Javascript. Each of these languages has a different purpose in building a website. Look at the following illustration.

HTML

HTML provides the structure or foundation of a web page. The web page cannot exist without HTML. HTML describes which components (i.e buttons, menus, text fields, etc), images, and data will be displayed.

CSS

CSS provides the design of a web page. How it will look. Things like colors, font size, spacing, and positioning of elements on the page are all the job of CSS.

Javascript

Javascript provides the functionality of the web page. HTML provides the button, but it is Javascript that makes the button do something when clicked.

document.getElementById("reset").addEventListener("click", clear);

function convert()
{
   var ans;
   var value = document.getElementById("num").value;
   var radio = getRadioVal( document.getElementById('group'), 'Converter' );
   if(radio == "decimal")
      ans = decimalToBinary(value);
   else
      ans = binaryToDecimal(value);

   document.getElementById("ans").value = ans;
}

function clear()
{
   document.getElementById("num").value = "";
   document.getElementById("ans").value = "";
}

function getRadioVal(form, name) 
{
   var val;
   // get list of radio buttons with specified name
   var radios = form.elements[name];

   // loop through list of radio buttons
   var len=radios.length;
   for (let i=0; i<len; i++) 
   {
	  if ( radios[i].checked ) 
      { 
         val = radios[i].value; // if so, hold its value in val
         break; // and break out of for loop
      }
    }
    return val; // return value of checked radio or undefined if none checked
}

function binaryToDecimal(value)
{
   value = value+"";
   var place = 1;
   var ans = 0;
   for(let i=value.length - 1; i >= 0; i--)
   {
      let bit = value.substring(i, i+1);
      if(bit == 1)
      {
         ans += place;
      }
      place = place * 2;
   }
   return ans;
}

function decimalToBinary(value)
{
   let n = parseInt(value);
   return n.toString(2);
}
